// 3.2 - Blink a 2 LED using Switch.


int led1 = 6;
int led2 = 7;
int Switch = 5;
void setup()
{
  pinMode(led1,OUTPUT);
  pinMode(led2,OUTPUT);
  pinMode(Switch,INPUT);
}

void loop()
{
 int state = digitalRead(Switch);
 
  if(state==HIGH)
  {
    digitalWrite(led1,HIGH);
    digitalWrite(led2,HIGH);
  }
  else
  {
    digitalWrite(led1,LOW);
    digitalWrite(led2,LOW);
  }
}
