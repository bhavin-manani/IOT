// 3.1 - Blink an LED with Push Button using Arduino. 

int pushbutton = 2;
int LED = 10;
void setup()
{
  pinMode(pushbutton,INPUT);
  pinMode(LED,OUTPUT);
}

void loop()
{
  int btnState = digitalRead(pushbutton);
  if(btnState == HIGH)
  {
    digitalWrite(LED,HIGH);
    delay(1000);
    digitalWrite(LED,LOW);
    delay(1000);
  }
  else
  {
     digitalWrite(LED,LOW);
  }
}
