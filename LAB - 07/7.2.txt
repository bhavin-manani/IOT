// 7.2 - Control an LED  via PIR sensor using Arduino
//
int LED=13;
int PIR=5;

void setup()
{
  pinMode(LED, OUTPUT);
  pinMode(PIR, INPUT);
}

void loop()
{
  int Motion = digitalRead(PIR);
  if(Motion==HIGH)
  {
    digitalWrite(LED,HIGH);
  }
  else
  {
    digitalWrite(LED,LOW);
  }
  
}
