// 4.3 -  Perform a traffic light signal with Red, Yellow and Green LEDs. (A)


void setup() 
{
  pinMode(3, OUTPUT); 
  pinMode(4, OUTPUT);  
  pinMode(5, OUTPUT); 
}

void loop() 
{
//RED
  digitalWrite(3, HIGH);
  delay(5000); 
  digitalWrite(3, LOW);
  
//Yellow
  digitalWrite(4, HIGH);
  delay(2000);
  digitalWrite(4, LOW);
   
//Green
  digitalWrite(5, LOW);
  delay(2000);
  digitalWrite(5, HIGH);
   
}
